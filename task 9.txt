#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 1000; i++) {
        int tersi = 0, n = i;

        while (n > 0) {
            int q = n % 10;
            tersi = tersi * 10 + q;
            n = n / 10;
        }

        if (i == tersi)
            cout << i << " ";
    }

    return 0;
}
